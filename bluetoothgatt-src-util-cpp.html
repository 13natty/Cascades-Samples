    const char* Util::parse_uuid(const char *uuid, const uuidnames_t uuidNames[])
    {
        uint16_t id = 0;

        if (0 == uuid)
            return "UNKNOWN";

        if (strncasecmp(uuid, "0x", 2) == 0)
            uuid += 2;

        if (strlen(uuid) != 4)
            return uuid;

        if (sscanf(uuid, "%04hx", &id) != 1)
            return uuid;

        for (int i = 0; uuidNames[i].uuid != -1; i++) {
            if (uuidNames[i].uuid == id) {
                return uuidNames[i].name;
            }
        }

        return uuid;
    }
