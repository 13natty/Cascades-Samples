    import bb.cascades 1.0

    Container {
        id: root
        property alias x: position.positionX
        property alias y: position.positionY
        property double lat
        property double lon
        property alias anim: anim

        layoutProperties: AbsoluteLayoutProperties {
            id: position
            onPositionXChanged: {
                root.remove(bubble);
            }
            onPositionYChanged: {
                root.remove(bubble);
            }
        }

        ImageView {
            animations: [
                ScaleTransition {
                    id: anim
                    fromX: 0
                    toX: 1
                    fromY: 0
                    toY: 1
                    duration: 500
                    easingCurve: StockCurve.BounceOut
                }
            ]
            imageSource: "asset:///images/pin.png"
            focusPolicy: FocusPolicy.Touch
            onFocusedChanged: {
                if (focused) {
                    anim.play();
                    root.add(bubble);
                } else {
                    root.remove(bubble);
                }
            }
        }

        attachedObjects: [
            Bubble {
                id: bubble
            }
        ]
        overlapTouchPolicy: OverlapTouchPolicy.Allow
    }
